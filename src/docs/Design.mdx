import { Meta, Title, Subtitle } from "@storybook/addon-docs/blocks"
import designSource from "../../DESIGN.md?raw"
import { ColorTokenSwatches } from "@/components/ui/_storybook/ColorTokenSwatches"
import { DesignMarkdown } from "@/components/ui/_storybook/DesignMarkdown"
import "./design-doc.css"

export const designDocParts = (() => {
  const marker = "## Visual language"
  const split = designSource.indexOf(marker)
  if (split === -1) {
    return { before: designSource, fromVisualLanguage: "" }
  }
  return {
    before: designSource.slice(0, split).trim(),
    fromVisualLanguage: designSource.slice(split).trim(),
  }
})()

<Meta
  title="Docs/Design System"
  parameters={{
    layout: "fullscreen",
    docs: {
      canvas: { hidden: true },
      toc: {
        headingSelector:
          ".sb-design-doc-content h2, .sb-design-doc-content h3, .sb-color-swatches h2, .sb-color-swatches h3",
        title: "On this page",
      },
    },
  }}
/>

<Title>Design System</Title>

<Subtitle>
  Live view of <code>DESIGN.md</code> — tokens, themes, color &amp; visual language, typography,
  and component guidance.
</Subtitle>

<DesignMarkdown source={designDocParts.before} />

<div className="sb-color-swatches">
  <ColorTokenSwatches />
</div>

{designDocParts.fromVisualLanguage ? (
  <DesignMarkdown source={designDocParts.fromVisualLanguage} />
) : null}
